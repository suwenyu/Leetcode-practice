### Others Solution
```c++
class Solution {
public:
    int subarraySum(vector<int>& nums, int k) {
        unordered_map<int, int> prevSum; 
  
        int res = 0; 
        // Sum of elements so far. 
        int currsum = 0; 
        int n = nums.size();
            
        for (int i = 0; i < n; i++) { 

            currsum += nums[i]; 

            if (currsum == k)  
                res++;         

            if (prevSum.find(currsum - k) !=  
                                      prevSum.end())  
                res += (prevSum[currsum - k]); 

            prevSum[currsum]++; 
        } 

        return res; 
    }
};
```
